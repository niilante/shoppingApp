				function Palindrome(str) { 

				  str = str.replace(/ /g,"").toLowerCase();
				  var compareStr = str.split("").reverse().join("");

				  if (compareStr === str) {
				    return true;
				  } 
				  else {
				    return false;
				  } 

				}

				function rotate(arr, steps){
				var rotations = steps;
				var index;

				while(rotations > 0){
					index = arr.pop();
					arr.unshift(index);

					rotations--;
				}

				return arr;
				}

			// EXAMPLES

			rotate([22,11,500,92], 2);
			// --> returns [ 500, 92, 22, 11 ];

			// rotate([11,15,28,9,10], 0)
			// --> returns [11,15,28,9,10]
